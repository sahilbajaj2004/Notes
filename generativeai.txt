1. INTRODUCTION TO GENERATIVE AI
================================================================================

WHAT IS GENERATIVE AI?
- AI that can create new content (text, images, code, audio)
- Uses neural networks trained on vast datasets
- Learns patterns to generate human-like outputs
- Examples: ChatGPT, DALL-E, Midjourney, GitHub Copilot

KEY CONCEPTS:
- Large Language Models (LLMs): AI trained on text data
- Tokens: Units of text (words, subwords, characters)
- Context Window: Maximum input tokens model can process
- Temperature: Controls randomness (0=deterministic, 1=creative)
- Prompts: Input instructions to the AI
- Embeddings: Vector representations of text

POPULAR AI PROVIDERS:
- OpenAI: GPT-4, GPT-3.5, DALL-E, Whisper
- Anthropic: Claude
- Google: Gemini (formerly Bard), PaLM
- Meta: LLaMA
- Cohere: Command, Embed
- Hugging Face: Open-source models
- Mistral AI: Mistral, Mixtral

USE CASES IN WEB DEVELOPMENT:
- Chatbots and virtual assistants
- Content generation (articles, descriptions)
- Code generation and completion
- Image generation from text
- Text summarization
- Language translation
- Sentiment analysis
- Search and recommendations
- Document Q&A

    conversationHistory.push({ role: 'user', content: userMessage });
    
    const completion = await openai.chat.completions.create({
        model: 'gpt-4',
        messages: conversationHistory,

    conversationHistory.push({ role: 'assistant', content: assistantMessage });

PARAMETERS:
model          - AI model to use (gpt-4, gpt-3.5-turbo)
messages       - Array of conversation messages
temperature    - Randomness (0-2, default 1)
max_tokens     - Maximum response length
top_p          - Nucleus sampling (alternative to temperature)
frequency_penalty - Reduce repetition (-2 to 2)
presence_penalty  - Encourage new topics (-2 to 2)
stop           - Stop sequences
stream         - Enable streaming responses

MESSAGE ROLES:
system    - Sets behavior/personality of AI
user      - User's input
assistant - AI's response
function  - Function call results (deprecated, use tools)
tool      - Tool call results

3. STREAMING RESPONSES
================================================================================

WHY STREAMING?
- Better user experience for long responses
- Shows content as it's generated
- Reduces perceived latency

EXPRESS STREAMING ENDPOINT:
const express = require('express');
const OpenAI = require('openai');

const app = express();
const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

app.post('/api/chat/stream', async (req, res) => {
    const { message } = req.body;
    
    res.setHeader('Content-Type', 'text/event-stream');
    res.setHeader('Cache-Control', 'no-cache');
    res.setHeader('Connection', 'keep-alive');
    
    try {
        const stream = await openai.chat.completions.create({
            model: 'gpt-4',
            messages: [
                { role: 'system', content: 'You are a helpful assistant.' },
                { role: 'user', content: message }
            ],
            stream: true
        });
        
        for await (const chunk of stream) {
            const content = chunk.choices[0]?.delta?.content || '';
            if (content) {
                res.write(`data: ${JSON.stringify({ content })}\n\n`);
            }
        }
        
        res.write('data: [DONE]\n\n');
        res.end();
    } catch (error) {
        res.write(`data: ${JSON.stringify({ error: error.message })}\n\n`);
        res.end();
    }
});

REACT STREAMING CLIENT:
function ChatComponent() {
    const [response, setResponse] = useState('');
    const [loading, setLoading] = useState(false);
    
    const sendMessage = async (message) => {
        setLoading(true);
        setResponse('');
        
        try {
            const res = await fetch('/api/chat/stream', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ message })
            });
            
            const reader = res.body.getReader();
            const decoder = new TextDecoder();
            
            while (true) {
                const { done, value } = await reader.read();
                if (done) break;
                
                const chunk = decoder.decode(value);
                const lines = chunk.split('\n');
                
                for (const line of lines) {
                    if (line.startsWith('data: ')) {
                        const data = line.slice(6);
                        if (data === '[DONE]') continue;
                        
                        try {
                            const parsed = JSON.parse(data);
                            if (parsed.content) {
                                setResponse(prev => prev + parsed.content);
                            }
                        } catch (e) {}
                    }
                }
            }
        } catch (error) {
            console.error('Stream error:', error);
        } finally {
            setLoading(false);
        }
    };
    
    return (
        <div>
            <button onClick={() => sendMessage('Explain React hooks')}>
                Ask Question
            </button>
            {loading && <p>Generating...</p>}
            <div className="response">{response}</div>
        </div>
    );
}

// Chat with function calling
async function chatWithTools(userMessage) {
    const messages = [
        { role: 'system', content: 'You are a helpful assistant.' },
        { role: 'user', content: userMessage }
    ];
    
    // Initial request
    let response = await openai.chat.completions.create({
        model: 'gpt-4',
        messages,
        tools,
        tool_choice: 'auto'  // or 'none', or specific function
    });
    
    let message = response.choices[0].message;
    
    // Process tool calls
    while (message.tool_calls) {
        const toolCalls = message.tool_calls;
        messages.push(message);
        
        // Execute each tool call
        for (const toolCall of toolCalls) {
            const functionName = toolCall.function.name;
            const args = JSON.parse(toolCall.function.arguments);
            
            console.log(`Calling function: ${functionName}`, args);
            
            const result = await functions[functionName](args);
            
            messages.push({
                role: 'tool',
                tool_call_id: toolCall.id,
                content: result
            });
        }
        
        // Get next response
        response = await openai.chat.completions.create({
            model: 'gpt-4',
            messages,
            tools
        });
        
        message = response.choices[0].message;
    }
    
    return message.content;
}

// Usage

EMBEDDINGS:
- Convert text to numerical vectors
- Similar texts have similar vectors
- Used for semantic search, recommendations
- OpenAI: text-embedding-3-small, text-embedding-3-large

CREATING EMBEDDINGS:
async function createEmbedding(text) {
    const response = await openai.embeddings.create({
        model: 'text-embedding-3-small',
        input: text
    });
    
    return response.data[0].embedding;  // Array of floats
}

// Embed multiple texts
async function createEmbeddings(texts) {
    const response = await openai.embeddings.create({
        model: 'text-embedding-3-small',
        input: texts  // Array of strings
    });
    
    return response.data.map(item => item.embedding);
}

COSINE SIMILARITY:
function cosineSimilarity(vecA, vecB) {
    let dotProduct = 0;
    let normA = 0;
    let normB = 0;
    
    for (let i = 0; i < vecA.length; i++) {
        dotProduct += vecA[i] * vecB[i];
        normA += vecA[i] * vecA[i];
        normB += vecB[i] * vecB[i];
    }
    
    return dotProduct / (Math.sqrt(normA) * Math.sqrt(normB));
}

// Semantic search (basic - for small datasets)
async function semanticSearch(query, limit = 5) {
    const queryEmbedding = await createEmbedding(query);
    const documents = await Document.find({});
    
    const results = documents
        .map(doc => ({
            document: doc,
            score: cosineSimilarity(queryEmbedding, doc.embedding)
        }))
        .sort((a, b) => b.score - a.score)
        .slice(0, limit);
    
    return results;
}

// MongoDB Atlas Vector Search (production)
async function vectorSearch(query, limit = 5) {
    const queryEmbedding = await createEmbedding(query);
    
    const results = await Document.aggregate([
        {
            $vectorSearch: {
                index: 'vector_index',
                path: 'embedding',
                queryVector: queryEmbedding,
                numCandidates: 100,
                limit: limit
            }
        },
        {
            $project: {
                title: 1,
                content: 1,
                score: { $meta: 'vectorSearchScore' }
            }
        }
    ]);
    
    return results;
}

7. RAG (RETRIEVAL AUGMENTED GENERATION)
================================================================================

WHAT IS RAG?
- Combines retrieval with generation
- Retrieves relevant documents
- Uses them as context for AI
- Reduces hallucinations
- Enables knowledge-base Q&A

RAG WORKFLOW:
1. Ingest documents → split into chunks
2. Create embeddings for chunks
3. Store in vector database
4. User asks question
5. Embed question, find similar chunks
6. Pass chunks as context to LLM
7. LLM generates answer using context

async function ragQuery(question) {
    // 1. Retrieve relevant documents
    const docs = await vectorStore.similaritySearch(question, 4);
    
    // 2. Format context
    const context = docs.map(doc => doc.pageContent).join('\n\n');
    
    // 3. Create prompt with context
    const prompt = `Use the following context to answer the question.
If the answer is not in the context, say "I don't have that information."

Context:
${context}

Question: ${question}

Answer:`;
    
    // 4. Generate response
    const response = await openai.chat.completions.create({
        model: 'gpt-4',
        messages: [
            { role: 'system', content: 'You are a helpful assistant.' },
            { role: 'user', content: prompt }
        ],
        temperature: 0.3  // Lower for factual responses
    });
    
    return {
        answer: response.choices[0].message.content,
        sources: docs.map(doc => ({
            content: doc.pageContent.substring(0, 200),
            metadata: doc.metadata
        }))
    };
}

// backend/services/ragService.js
const { OpenAIEmbeddings, ChatOpenAI } = require('@langchain/openai');
const { RecursiveCharacterTextSplitter } = require('langchain/text_splitter');
const { MemoryVectorStore } = require('langchain/vectorstores/memory');

class RAGService {
    constructor() {
        this.embeddings = new OpenAIEmbeddings();
        this.model = new ChatOpenAI({ 
            modelName: 'gpt-4',
            temperature: 0.3
        });
        this.vectorStore = null;
        this.conversationHistory = [];
    }
    
    async ingestDocuments(documents) {
        const splitter = new RecursiveCharacterTextSplitter({
            chunkSize: 1000,
            chunkOverlap: 200
        });
        
        const chunks = await splitter.splitDocuments(documents);
        
        this.vectorStore = await MemoryVectorStore.fromDocuments(
            chunks,
            this.embeddings
        );
        
        return chunks.length;
    }
    
    async query(question) {
        if (!this.vectorStore) {
            throw new Error('No documents ingested');
        }
        
        // Retrieve relevant context
        const docs = await this.vectorStore.similaritySearch(question, 4);
        const context = docs.map(d => d.pageContent).join('\n\n---\n\n');
        
        // Build messages with history
        const messages = [
            {
                role: 'system',
                content: `You are a helpful assistant that answers questions based on the provided context.
                
Context:
${context}

Instructions:
- Answer based on the context provided
- If the answer isn't in the context, say so
- Be concise but thorough
- Cite relevant parts of the context`
            },
            ...this.conversationHistory.slice(-6),  // Last 3 exchanges
            { role: 'user', content: question }
        ];
        
        const response = await this.model.invoke(messages);
        
        // Update history
        this.conversationHistory.push(
            { role: 'user', content: question },
            { role: 'assistant', content: response.content }
        );
        
        return {
            answer: response.content,
            sources: docs.map(d => ({
                content: d.pageContent.substring(0, 150) + '...',
                metadata: d.metadata
            }))
        };
    }
    
    clearHistory() {
        this.conversationHistory = [];
    }
}

module.exports = new RAGService();

function RAGChatbot() {

                {messages.map((msg, i) => (

                        <p>{msg.content}</p>
                        {msg.sources && (
                            <div className="sources">
                                <small>Sources:</small>
                                {msg.sources.map((s, j) => (
                                    <p key={j}>{s.content}</p>
                                ))}
                            </div>
                        )}
                    </div>
                ))}
            </div>
            
            <div className="input-section">
                <input
                    value={input}
                    onChange={(e) => setInput(e.target.value)}
                    onKeyPress={(e) => e.key === 'Enter' && sendMessage()}
                    placeholder="Ask a question..."
                    disabled={loading}
                />
                <button onClick={sendMessage} disabled={loading}>
                    {loading ? 'Thinking...' : 'Send'}
                </button>
            </div>
        </div>
    );
}

ERROR HANDLING:
async function callOpenAI(messages, retries = 3) {
    for (let attempt = 1; attempt <= retries; attempt++) {
        try {
            const response = await openai.chat.completions.create({
                model: 'gpt-4',
                messages
            });
            return response.choices[0].message.content;
        } catch (error) {
            // Rate limit error
            if (error.status === 429) {
                const waitTime = Math.pow(2, attempt) * 1000;
                console.log(`Rate limited. Waiting ${waitTime}ms...`);
                await new Promise(r => setTimeout(r, waitTime));
                continue;
            }
            
            // Context length exceeded
            if (error.code === 'context_length_exceeded') {
                throw new Error('Message too long. Please shorten your input.');
            }
            
            // Invalid API key
            if (error.status === 401) {
                throw new Error('Invalid API key');
            }
            
            // Server error - retry
            if (error.status >= 500) {
                if (attempt < retries) {
                    await new Promise(r => setTimeout(r, 1000 * attempt));
                    continue;
                }
            }
            
            throw error;
        }
    }
}

// Token-based rate limiting
const tokenBucket = new Map();

function checkTokenLimit(userId, tokensUsed) {
    const now = Date.now();
    const userBucket = tokenBucket.get(userId) || { tokens: 0, resetAt: now + 3600000 };
    
    if (now > userBucket.resetAt) {
        userBucket.tokens = 0;
        userBucket.resetAt = now + 3600000;
    }
    
    userBucket.tokens += tokensUsed;
    tokenBucket.set(userId, userBucket);
    
    const limit = 100000;  // 100k tokens per hour
    if (userBucket.tokens > limit) {
        throw new Error('Token limit exceeded');
    }
    
    return limit - userBucket.tokens;
}

function calculateCost(model, usage) {
    const pricing = {
        'gpt-4': { prompt: 0.03, completion: 0.06 },
        'gpt-3.5-turbo': { prompt: 0.0015, completion: 0.002 }
    };
    
    const rates = pricing[model] || pricing['gpt-3.5-turbo'];
    return (
        (usage.prompt_tokens / 1000) * rates.prompt +
        (usage.completion_tokens / 1000) * rates.completion
    );
}

async function cachedCompletion(prompt, options = {}) {
    const cacheKey = crypto
        .createHash('md5')
        .update(JSON.stringify({ prompt, ...options }))
        .digest('hex');
    
    // Check cache
    const cached = await redis.get(cacheKey);
    if (cached) {
        return JSON.parse(cached);
    }
    
    // Call API
    const response = await openai.chat.completions.create({
        model: options.model || 'gpt-4',
        messages: [{ role: 'user', content: prompt }],
        ...options
    });
    
    const result = response.choices[0].message.content;
    
    // Cache for 1 hour
    await redis.setex(cacheKey, 3600, JSON.stringify(result));
    
    return result;
}
